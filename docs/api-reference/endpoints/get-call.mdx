---
title: "Get Call"
api: "GET /get-call"
description: "Retrieve details of a specific call including transcript and recording"
---

Get detailed information about a call, including its status, transcript, summary, and recording URL.

## Request Body

<ParamField body="call_id" type="string" required>
  The unique identifier of the call to retrieve.
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique call identifier.
</ResponseField>

<ResponseField name="agent_id" type="string">
  The agent that handled this call.
</ResponseField>

<ResponseField name="from_number" type="string">
  The phone number that made the call.
</ResponseField>

<ResponseField name="to_number" type="string">
  The destination phone number.
</ResponseField>

<ResponseField name="direction" type="string">
  Either `inbound` or `outbound`.
</ResponseField>

<ResponseField name="status" type="string">
  Final call status: `completed`, `busy`, `no-answer`, `failed`, or `canceled`.
</ResponseField>

<ResponseField name="duration" type="integer">
  Call duration in seconds.
</ResponseField>

<ResponseField name="recording_url" type="string">
  Signed URL to download the call recording. Expires after 1 hour.
</ResponseField>

<ResponseField name="transcript" type="string">
  Full transcript of the conversation.
</ResponseField>

<ResponseField name="transcript_segments" type="array">
  Timestamped transcript segments with speaker identification.

  ```json
  [
    {
      "speaker": "agent",
      "text": "Hello, this is Sarah from Acme Dental.",
      "start_time": 0.5,
      "end_time": 3.2
    },
    {
      "speaker": "caller",
      "text": "Hi Sarah, I'm calling about my appointment.",
      "start_time": 3.5,
      "end_time": 6.1
    }
  ]
  ```
</ResponseField>

<ResponseField name="sentiment" type="string">
  AI-analyzed call sentiment: `positive`, `neutral`, or `negative`.
</ResponseField>

<ResponseField name="call_summary" type="string">
  AI-generated summary of the call.
</ResponseField>

<ResponseField name="metadata" type="object">
  Custom metadata attached when the call was created.
</ResponseField>

<ResponseField name="started_at" type="string">
  When the call was answered.
</ResponseField>

<ResponseField name="ended_at" type="string">
  When the call ended.
</ResponseField>

```javascript Node.js
const response = await fetch(
  'https://mtxbiyilvgwhbdptysex.supabase.co/functions/v1/get-call',
  {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_API_TOKEN',
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({

  }
```

```python Python
import requests

response = requests.post(
    'https://mtxbiyilvgwhbdptysex.supabase.co/functions/v1/get-call',
    headers={
        'Authorization': 'Bearer YOUR_API_TOKEN',
        'Content-Type': 'application/json',
    },
    json={'call_id': '7f3d8e00-a1b2-4c3d-9e4f-567890abcdef'}
)

call = response.json()
print(f"Summary: {call['call_summary']}")
```
