---
title: "Get Custom Function"
api: "GET /custom-functions/{id}"
description: "Get a specific custom function by ID"
---

Retrieve details of a specific custom function.

## Path Parameters

<ParamField path="id" type="string" required>
  UUID of the custom function
</ParamField>

## Response

<ResponseField name="id" type="string">
  UUID of the function
</ResponseField>

<ResponseField name="agent_id" type="string">
  UUID of the agent this function belongs to
</ResponseField>

<ResponseField name="user_id" type="string">
  UUID of the user who created this function
</ResponseField>

<ResponseField name="name" type="string">
  Function name (snake_case)
</ResponseField>

<ResponseField name="description" type="string">
  Function description used by AI to decide when to call
</ResponseField>

<ResponseField name="http_method" type="string">
  HTTP method (GET, POST, PUT, PATCH, DELETE)
</ResponseField>

<ResponseField name="endpoint_url" type="string">
  Webhook URL
</ResponseField>

<ResponseField name="headers" type="array">
  Custom headers included with requests
</ResponseField>

<ResponseField name="body_schema" type="array">
  Parameter definitions for the function
</ResponseField>

<ResponseField name="response_variables" type="array">
  Variables extracted from response
</ResponseField>

<ResponseField name="timeout_ms" type="integer">
  Request timeout in milliseconds
</ResponseField>

<ResponseField name="max_retries" type="integer">
  Number of retry attempts
</ResponseField>

<ResponseField name="is_active" type="boolean">
  Whether the function is active
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 last update timestamp
</ResponseField>

```bash cURL
  -H "Authorization: Bearer YOUR_API_KEY"
```
</RequestExample>
